{{- /* Shared product-link builder: canonical /dp/{asin} form with affiliate tag when configured. Used by the gift-guide product card AND the ItemList schema so both stay in sync. */ -}} {{- $p := .product -}} {{- $tag := .tag -}} {{- $u := $p.url -}} {{- with $p.asin -}} {{- $u = printf "https://www.amazon.com/dp/%s" . -}} {{- with $tag -}}{{- $u = printf "%s?tag=%s" $u . -}}{{- end -}} {{- else -}} {{- with $tag -}} {{- $sep := "?" -}} {{- if strings.Contains $u "?" }}{{- $sep = "&" }}{{- end -}} {{- $u = printf "%s%stag=%s" $u $sep $tag -}} {{- end -}} {{- end -}} {{- return $u -}}
